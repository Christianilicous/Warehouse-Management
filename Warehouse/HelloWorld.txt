CREATE TABLE management (
    product_id INT AUTO_INCREMENT PRIMARY KEY,  -- Unique identifier for each product
    product_name VARCHAR(255) NOT NULL,         -- Name of the product
    category VARCHAR(255),                      -- Category of the product
    stock_level INT NOT NULL DEFAULT 0,         -- Quantity available in stock
    location VARCHAR(255),                      -- Physical location of the product
    stock_status ENUM('In Stock', 'Low Stock', 'Out of Stock') NOT NULL DEFAULT 'In Stock', -- Stock status
    last_updated DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, -- Last update time
    tracking_info JSON                           -- Real-time tracking details
);
